    <script>
        let sidebarCollapsed = false;

        function toggleSidebar() {
            const sidebar = document.getElementById('sidebar');
            const mainContent = document.getElementById('main-content');
            const sidebarTexts = document.querySelectorAll('.sidebar-text');
            
            sidebarCollapsed = !sidebarCollapsed;
            
            if (sidebarCollapsed) {
                sidebar.style.width = '80px';
                mainContent.style.marginLeft = '80px';
                sidebarTexts.forEach(text => text.style.display = 'none');
            } else {
                sidebar.style.width = '256px';
                mainContent.style.marginLeft = '256px';
                sidebarTexts.forEach(text => text.style.display = 'block');
            }
        }

        function logout() {
            const modal = document.createElement('div');
            modal.className = 'fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50';
            modal.innerHTML = `
                <div class="glass-effect p-6 rounded-2xl shadow-2xl max-w-sm w-full mx-4">
                    <div class="text-center">
                        <div class="w-16 h-16 bg-gradient-to-br from-red-500 to-red-600 rounded-full flex items-center justify-center mx-auto mb-4">
                            <span class="text-white text-2xl">üö™</span>
                        </div>
                        <h3 class="text-xl font-bold mb-2">Confirm Logout</h3>
                        <p class="text-gray-400 mb-6">Are you sure you want to sign out of your admin session?</p>
                        <div class="flex space-x-3">
                            <button onclick="this.closest('.fixed').remove()" class="flex-1 px-4 py-2 bg-gray-600 text-white rounded-xl hover:bg-gray-700 transition-colors">
                                Cancel
                            </button>
                            <button onclick="window.location.href='#'" class="flex-1 px-4 py-2 gradient-bg text-white rounded-xl hover:shadow-lg transition-all">
                                Logout
                            </button>
                        </div>
                    </div>
                </div>
            `;
            document.body.appendChild(modal);
        }

        // Animate numbers on page load
        function animateNumber(element, target, duration = 2000) {
            const start = 0;
            const increment = target / (duration / 16);
            let current = start;
            
            const timer = setInterval(() => {
                current += increment;
                if (current >= target) {
                    current = target;
                    clearInterval(timer);
                }
                
                if (element.id === 'total-volume') {
                    element.textContent = '$' + (current / 1000000).toFixed(1) + 'M';
                } else if (element.id === 'total-users') {
                    element.textContent = Math.floor(current).toLocaleString();
                } else {
                    element.textContent = Math.floor(current).toLocaleString();
                }
            }, 16);
        }

        // Populate recent activity
        function populateRecentActivity() {
            const activities = [
                { type: 'user', message: 'New user registered: trader_mike23', time: '2 minutes ago', icon: 'üë§' },
                { type: 'trade', message: 'Large trade executed: $50,000 BTC/USD', time: '5 minutes ago', icon: 'üí±' },
                { type: 'alert', message: 'Price alert triggered for ETH', time: '8 minutes ago', icon: 'üö®' },
                { type: 'system', message: 'System backup completed successfully', time: '15 minutes ago', icon: 'üíæ' },
                { type: 'news', message: 'News article published: Bitcoin Rally', time: '23 minutes ago', icon: 'üì∞' }
            ];

            const container = document.getElementById('recent-activity');
            container.innerHTML = activities.map(activity => `
                <div class="flex items-center p-3 bg-gray-800/30 rounded-lg hover:bg-gray-700/30 transition-colors">
                    <div class="w-10 h-10 bg-primary/20 rounded-lg flex items-center justify-center mr-3">

                        <p class="text-xs text-gray-500">${activity.time}</p>

                </div>
            `).join('');
        }

        // Update stats periodically
        function updateStats() {
            const totalUsers = document.getElementById('total-users');
            const activeTraders = document.getElementById('active-traders');
            const totalVolume = document.getElementById('total-volume');
            
            // Simulate real-time updates
            const currentUsers = parseInt(totalUsers.textContent.replace(/,/g, ''));
            const currentTraders = parseInt(activeTraders.textContent.replace(/,/g, ''));
            const currentVolume = parseFloat(totalVolume.textContent.replace('$', '').replace('M', ''));
            
            totalUsers.textContent = (currentUsers + Math.floor(Math.random() * 10)).toLocaleString();
            activeTraders.textContent = (currentTraders + Math.floor(Math.random() * 5)).toLocaleString();
            totalVolume.textContent = '$' + (currentVolume + Math.random() * 0.5).toFixed(1) + 'M';
        }

        // Initialize dashboard
        document.addEventListener('DOMContentLoaded', function() {
            // Animate initial numbers
            setTimeout(() => {
                animateNumber(document.getElementById('total-users'), 8547);
                animateNumber(document.getElementById('active-traders'), 2341);
            }, 500);

            // Populate recent activity
            populateRecentActivity();

            // Update stats every 10 seconds
            setInterval(updateStats, 10000);

            // Add fade-in animation to cards
            const cards = document.querySelectorAll('.glass-effect');
            cards.forEach((card, index) => {
